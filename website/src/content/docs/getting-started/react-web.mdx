import { Steps, Tabs, TabItem } from '@astrojs/starlight/components';

<Steps>

1. Set up project via: `npx tiged --mode=git https://github.com/livestorejs/livestore/examples/linearlite my-app`

2. Install dependencies

  <Tabs syncKey="package-manager">
    <TabItem label="bun">`bun install`</TabItem>
    <TabItem label="pnpm">`pnpm install`</TabItem>
    <TabItem label="npm">`npm install`</TabItem>
    <TabItem label="yarn">`yarn install`</TabItem>
  </Tabs>

3. Run dev environment

  <Tabs syncKey="package-manager">
    <TabItem label="bun">`bun dev`</TabItem>
    <TabItem label="pnpm">`pnpm dev`</TabItem>
    <TabItem label="npm">`npm run dev`</TabItem>
    <TabItem label="yarn">`yarn dev`</TabItem>
  </Tabs>

4. Open browser

  Open `http://localhost:60000` in your browser.

  You can also open the devtools by going to `http://localhost:60000/_devtools.html`.

</Steps>
